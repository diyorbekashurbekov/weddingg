Structured Query Language

Деректер қорымен жұмыс жасауға арналған стандартты тіл. Деректерді сұрату, өзгерту, қосу, жою — барлығын SQL арқылы орындаймыз.

SELECT INSERT UPDATE DELETE JOIN WHERE GROUP BY

→ кілтін немесе стрелкені басыңыз

ҰБТ-дағы маңызы: SQL командалары тапсырмалары ҰБТ-да жиі кездеседі. SELECT, INSERT, UPDATE, DELETE операторларын жатқа білу керек.
SQL
-- Деректерді іріктеу
SELECT * FROM Students;

-- Жаңа жазба қосу
INSERT INTO Students
  VALUES (5, 'Ғалым', 17);

-- Деректерді жаңарту
UPDATE Students
  SET Age = 18
  WHERE ID = 1;

-- Жазбаны жою
DELETE FROM Students
  WHERE ID = 3;

SELECT операторы

Синтаксис
SELECT баған1, баған2
FROM кесте_аты
[WHERE шарт]
[ORDER BY баған ASC|DESC]
[LIMIT n];
Мысал
-- Барлық оқушыларды алу
SELECT * FROM Students;

-- Жасы 16-дан үлкен оқушылар
SELECT Name, Age
FROM Students
WHERE Age > 16;

-- Алфавит бойынша сұрыптау
SELECT * FROM Students
ORDER BY Name ASC;

Шарттар

WHERE шарты және операторлары

Салыстыру операторлары

=  тең != немесе <>  тең емес >  үлкен <  кіші >=  үлкен немесе тең <=  кіші немесе тең

Логикалық операторлар

  • AND Екі шарт та орындалу керек
  • OR Біреуі орындалса жеткілікті
  • NOT Шартты теріске шығару

Арнайы операторлар

  • BETWEEN Ауқымда: BETWEEN 15 AND 17
  • IN Тізімде: IN (1, 2, 3)
  • LIKE Үлгіге сәйкес: LIKE 'Ә%'
  • IS NULL Бос мән тексеру
Мысалдар
-- AND оператор
SELECT * FROM Students
WHERE Age >= 16 AND Class_ID = 2;

-- BETWEEN
SELECT * FROM Students
WHERE Age BETWEEN 15 AND 17;

-- LIKE — 'Ә' әрпінен басталатын
SELECT * FROM Students
WHERE Name LIKE 'Ә%';

-- IN — нақты мәндер тізімі
SELECT * FROM Students
WHERE Class_ID IN (1, 2);
LIKE ішіндегі шаблондар:
% — кез-келген символдар тізбегі
_ — бір ғана символ орны
DML операторлары

INSERT · UPDATE · DELETE

INSERT Жазба қосу
INSERT INTO Students
  (ID, Name, Age)
VALUES
  (5, 'Ғалым', 16);
Ережелер:
• Баған аттарын VALUES алдында жазу керек
• Мәтін мәндерді тырнақшаға алу керек
• Primary Key қайталанбауы керек
UPDATE Деректерді жаңарту
UPDATE Students
SET
  Age = 18,
  Name = 'Серіков'
WHERE ID = 1;
⚠ НАЗАР:
WHERE болмаса — барлық жолдар өзгереді! Бұл қауіпті қате.
DELETE Жазбаны жою
-- Бір жазбаны жою
DELETE FROM Students
WHERE ID = 3;

-- Барлығын тазарту
DELETE FROM Students;

-- немесе жылдам тазарту
TRUNCATE TABLE Students;
DELETE vs TRUNCATE:
DELETE — бір-бір жояды (WHERE болады)
TRUNCATE — барлығын жылдам тазартады

JOIN операторы

INNER JOIN Ең жиі

Екі кестеде де сәйкес жазбалар болғанда ғана қосады

LEFT JOIN

Сол кестенің барлық жолдарын + оң кестеден сәйкестерін алады

RIGHT JOIN

Оң кестенің барлық жолдарын + сол кестеден сәйкестерін алады

INNER JOIN мысалы

  Students.Name,
  Grades.Mark
FROM Students
INNER JOIN Grades
  ON Students.ID = Grades.Student_ID;

Ең үлкен / кіші мән

SELECT MAX(Mark) FROM Grades;
GROUP BY
-- Сынып бойынша орташа баға
SELECT
  Class_ID,
  AVG(Mark) AS Орташа_Баға,
  COUNT(*) AS Оқушылар_Саны
FROM Grades
GROUP BY Class_ID
HAVING AVG(Mark) > 4;

DDL — Кесте жасау

CREATE TABLE және деректер типтері

CREATE TABLE
CREATE TABLE Students (
  ID       INT PRIMARY KEY,
  Name    VARCHAR(100) NOT NULL,
  Age     INT CHECK(Age >= 6),
  Email   VARCHAR(150) UNIQUE,
  Class_ID INT DEFAULT 1,
  FOREIGN KEY (Class_ID)
    REFERENCES Classes(ID)
);
Кестені жою / өзгерту
DROP TABLE Students;
ALTER TABLE Students ADD Phone VARCHAR(20);
ALTER TABLE Students DROP COLUMN Phone;

Деректер типтері:

  • INT Бүтін сандар (1, 2, 100, -5)
  • VARCHAR(n) Айнымалы ұзындықты мәтін
  • CHAR(n) Тіркелген ұзындықты мәтін
  • FLOAT Бөлшек сандар (3.14, 2.5)
  • DATE Күн (YYYY-MM-DD)
  • BOOLEAN Логикалық мән (TRUE/FALSE)

Шектеулер (Constraints):

PRIMARY KEY FOREIGN KEY NOT NULL UNIQUE CHECK DEFAULT
